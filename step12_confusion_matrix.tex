\section*{Step 12 --- Confusion Matrix and Classification Report}

\begin{lstlisting}[style=pythonstyle]
pred_probs  = model.predict(test_data)
pred_labels = (pred_probs > 0.5).astype(int).reshape(-1)

true_labels = test_data.classes
class_names = list(test_data.class_indices.keys())

# --- Confusion Matrix ---
cm = confusion_matrix(true_labels, pred_labels)

plt.figure(figsize=(6, 5))
sns.heatmap(cm, annot=True, fmt='d',
            xticklabels=class_names,
            yticklabels=class_names,
            cmap='Blues')
plt.xlabel("Predicted Label")
plt.ylabel("True Label")
plt.title("Confusion Matrix")
plt.show()

# --- Classification Report ---
print(classification_report(true_labels, pred_labels,
                             target_names=class_names))
\end{lstlisting}

\subsection*{Explanation}

\begin{itemize}
    \item \texttt{model.predict(test\_data)} --- generates predicted probabilities for every test image.
    \item \texttt{pred\_labels = (pred\_probs > 0.5)} --- converts each probability into a class label (0 or 1).
    \item \textbf{Confusion matrix} --- shows how many images were classified correctly or incorrectly:
        \begin{itemize}
            \item Diagonal values = correct predictions.
            \item Off-diagonal values = misclassifications.
        \end{itemize}
    \item \textbf{Classification report} provides:
        \begin{itemize}
            \item \textbf{Precision} --- of all images predicted as class X, how many actually are X.
            \item \textbf{Recall} --- of all actual class X images, how many were correctly identified.
            \item \textbf{F1-score} --- harmonic mean of precision and recall.
        \end{itemize}
\end{itemize}
